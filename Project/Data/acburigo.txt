This worked: ``` import networkx as nx import matplotlib.pyplot as plt  graph = nx.Graph()  graph.add_nodes_from(['a', 'b', 'c']) graph.add_nodes_from(['1', '2', '3', '4', '5', '6', '7', '8', '9']) graph.add_edges_from([     ('a', '1'), ('a', '2'), ('a', '3'),     ('b', '4'), ('b', '5'), ('b', '6'),     ('c', '7'), ('c', '8'), ('c', '9'), ])  plt.figure(figsize=(20, 10)) pos = nx.drawing.spring_layout(graph)  nx.draw_networkx_nodes(     graph,     pos,     nodelist=['a', 'b', 'c'],     node_color=['r', 'g', 'b'],     node_shape='o', )  nx.draw_networkx_nodes(     graph,     pos,     nodelist=['1', '2', '3', '4', '5', '6', '7', '8', '9'],     node_color=['r', 'r', 'r', 'g', 'g', 'g', 'b', 'b', 'b'],     node_shape='v', )  nx.draw_networkx_edges(graph, pos) nx.draw_networkx_labels(graph, pos)  plt.show() ```  ![index](https://user-images.githubusercontent.com/20728609/69020018-e8458700-0991-11ea-9a74-927dd243d5f6.png)
